public class Order {
    private ArrayList<Product> products;

    public Order() {
        this.products = new ArrayList<>();
    }

    public void addProduct(Product product) {
        products.add(product);
        System.out.println("Added: "
            + product.getName());
    }

    public double getTotal() {
        double total = 0;
        for (Product p : products) {
            total += p.getPrice();
        }
        return total;
    }
}

class Order {
    constructor() {
        this.products = [];
    }

    addProduct(product) {
        this.products.push(product);
        this.updateOrderDisplay();
        this.showAddedAnimation(product);
    }

    getTotal() {
        return this.products.reduce(
            (sum, p) => sum + p.price, 0
        );
    }

    updateOrderDisplay() {
        document.getElementById('order-items')
            .innerHTML = this.renderItems();
    }
}

public void checkout() {
    System.out.println("=== RECEIPT ===");

    for (Product p : products) {
        System.out.println(
            p.getName() + " - $"
            + String.format("%.2f", p.getPrice())
        );
    }

    System.out.println("-------------");
    System.out.println("TOTAL: $"
        + String.format("%.2f", getTotal()));

    saveToFile();
}

function checkout() {
    const receipt = document.createElement('div');
    receipt.className = 'receipt-kiosk';

    receipt.innerHTML = `
        <h3>Your Order</h3>
        ${currentOrder.products.map(p => `
            <div class="receipt-item">
                <span>${p.name}</span>
                <span>$${p.price.toFixed(2)}</span>
            </div>
        `).join('')}
        <div class="receipt-total">
            TOTAL: $${currentOrder.getTotal()}
        </div>
    `;

    showScreen('checkout');
}
